            !isBeriplexAdministered &&
            isVitkAdministered != null &&
            !isVitkAdministered) {
            isEnabled = true;
        }

                isEnabled = true;
            }

            !isBeriplexAdministered &&
            isVitkAdministered != null &&
            isVitkAdministered) {
            if (vitkDate != null &&
                vitkTime != null) {
                isEnabled = true;

            if (beriplexDate != null &&
                beriplexTime != null &&
                vitkDate != null &&
                vitkTime != null) {

    }
